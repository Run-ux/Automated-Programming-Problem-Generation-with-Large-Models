    \textbf{算法 1} & \text{\textbf{正确的} Floyd-Warshall 算法} \\

    & M_{i,j}=\begin{cases}
                  0, & \text{当 } i=j \\

                  \infty, & \text{其他情况}

    2&\ \textbf{for } x=1,2,3,\ldots,n \textbf{ do} \\
    3&\ \qquad\textbf{for } y=1,2,3,\ldots,n \textbf{ do} \\

    5&\ \qquad\qquad\qquad M_{y,z}\gets \min(M_{y,z},M_{y,x}+M_{x,z}) \\
    \hline
    \end{array} 
$$
